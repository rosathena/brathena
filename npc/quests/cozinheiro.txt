			mes "Bem, quando você precisar";
			mes "comprar "+getitemname(getarg(0));

			close;
		}
		else if (.@sell > 100) {
			mes "[Madeleine Chu]";
			mes "Ah, desculpe, mas";
			mes "eu não vendo mais do que";
			mes "100 "+getitemname(getarg(0));
			mes "de uma vez, é apenas por segurança.";
			next;
		}
		else break;
	}
	set .@total_cost,.@sell * .@item_cost;
	set .@total_weight,.@sell * .@item_weight;
	if (Zeny < .@total_cost) {
		mes "[Madeleine Chu]";
		mes "Ah, desculpe, mas você";
		mes "você não pode comprar";
		mes getitemname(getarg(0));
		mes "Por favor verifique seus Zenys";

		close;

		mes "Desculpe, mas você não possui";
		mes "espaço suficiente em seu";
		mes "inventário para tantos";

		close;
	}
	set Zeny, Zeny-.@total_cost;
	getitem getarg(0),.@sell;
	mes "[Madeleine Chu]";
