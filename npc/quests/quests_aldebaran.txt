			mes "[Mercador]";
			mes "Que seja. Só não seja pego com isso.";
			getitem 2273,1;  // Espelho de Médico
			close;
		}
		else {
			mes "[Mercador]";
		  mes "Você cara!~";
		  mes "Verifique os requisitos de novo.";
		  mes "Você não parece idiota. Então... caia na real.";
			close;
		}

		if ((countitem(2247) > 0) && (countitem(916) > 299) && (Zeny > 499)) {
			delitem 2247,1;   // Chapéu Romântico
			delitem 916,300;  // Plumas

			mes "Ah! Você tem bom senso de moda. Eu sei que você teve um tempo difícil coletando esse material, mas este chapéu vale a pena. Pegue. Tudo que você precisa agora é de um casaco de pele e uma bengala!";

			mes "[Mercador]";
			mes "Ei cara~";

			close;
		}

			delitem 998,20;  // Ferro

		else {
			mes "[Mercador]";

		}
		else {
			mes "[Mercador]";
