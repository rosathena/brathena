	mes "[Estilista]";

	mes "[Estilista]";
	mes "Se você me der um Cupom de Tintura,";

	mes "Se você renascer, ^FF0000perderá o novo visual^000000";
	mes "e precisará de um novo Cupom de Tintura para pintar";

	next;
	mes "[Estilista]";
	mes "Hihihi~! Prontinho, recados dados...";

	next;

		switch(select("Abrir o catálogo de roupas:Tingir as roupas:Voltar a cor original:Cancelar.")) {
		case 1:
			mes "[Estilista]";
			mes "Venha ver que cores mais lindas.";
			mes "A primeira cor é a superior,";
			mes "a segunda é a inferior.";
			mes "Não se esqueça da sua cor";
			mes "favorita";

			for(set .@i,0; .@i < getarraysize(.@class$); set .@i, .@i + 1){

				mes "Superior: Primeira cor";
				mes "Inferior: Segunda cor";

				mes "barras de HP/SP na tela.";

				next;
			}
				cutin "",255;
				mes "[Estilista]";
				mes "Achou sua cor favorita?";
				next;

			mes "[Estilista]";

				mes "tudo bem?";
				set .@look,2;
				break;

				mes "[Estilista]";
				mes "Você escolheu a ^FF0000Segunda cor^000000, está";
				mes "tudo bem?";
				set .@look,3;
				break;
			case 3:
				set .@look,0;
			}
			if(!.@look)
				break;
			next;
			if(select("Mude minha roupa.:Vou deixar assim mesmo(Cancelar).") == 2) {
				mes "[Estilista]";
				mes "Bem, gosto é gosto, não é mesmo? Se mudar";
				mes "de idéia, é só aparecer!";
				close;
			}

				mes "[Estilista]";
				mes "Bela escolha! Você tem um bom olho";
				mes "para a alta-costura, não é mesmo?";
				next;

				mes "Prontinho! Viu só? Não doeu nada!";

			} else {
				mes "[Estilista]";
				mes "Hm, benzinho... Não esqueceu de nada não?";
				mes "Você precisa de um ^0000FFCupom de Tintura^000000";
				mes "para tingir suas roupas...";
				close;
			}
		case 3:
			if(getlook(7) == 0) {
				mes "[Estilista]";
				mes "Ops, você já está com a cor";
				mes "original na sua roupa.";
				mes "Tingir duas vezes não vai fazer diferença.";
				close;
			}
				mes "[Estilista]";
				mes "Tem certeza que deseja";
				mes "voltar a cor da roupa ao normal?";
				next;
			if(select("Mude minha roupa.:Vou deixar assim mesmo(Cancelar).") == 2) {
				mes "[Estilista]";
				mes "Bem, gosto é gosto, não é mesmo? Se mudar";
				mes "de idéia, é só aparecer!";
				close;
			}
			if(countitem(6295) > 0) {
				mes "[Estilista]";
				mes "*estalar de dedos*";
				mes "Prontinho! Viu só? Não doeu nada!";
				mes "Agora saia já daqui e desfile esta";
				mes "peça magnífica pelas passarelas de Prontera!";
				delitem 6295,1;
				setlook 7,0;
				close;
			} else {
				mes "[Estilista]";
				mes "Hm, benzinho... Não esqueceu de nada não?";
				mes "Você precisa de um ^0000FFCupom Removedor^000000";
				mes "para remover a tintura de suas roupas...";
				close;
			}
		case 4:
			mes "[Estilista]";
			mes "Ai, você não acha essa história de ''cor original''";
			mes "uma coisa muuuito ''retrô''?";
			mes "Bem, gosto é gosto, não é mesmo? Se mudar";
			mes "de idéia, é só aparecer!";
			close;
		}
	}
