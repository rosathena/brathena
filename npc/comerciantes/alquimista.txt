		break;

		break;

		break;

		callsub Venda_Manual,6285,100000;  // Livro de Farmacologia
		break;
	case 8:
		callsub Venda_Manual,11022,100000; // Livro de Culinária Avançada
		break;
	case 9:
		callsub Venda_Manual,11023,100000; // Como Aumentar sua Força

		mes "Por favor volte novamente.";

	}

Venda_Manual:
	mes "[Vendedor de Livros de Alquímia]";
	mes ""+getitemname(getarg(0))+"?";
	mes "Custará "+getarg(1)+" zeny cada.";
	mes "Quantos desse livro você gostaria de comprar?";
	next;
	mes "[Vendedor de Livros de Alquímia]";
	input .@amount;
	if (.@amount == 0) {
		mes "Você gostaria de ver alguns livros diferentes? Tudo bem.";
		close;
	}
	if (.@amount < 1 || .@amount > 99 ){
		mes "Você não pode comprar mais de 100 Livros de uma só vez.";
		close;
	}
	mes "Você escolheu "+.@amount+"x "+getitemname(getarg(0))+".";
	mes "Que custarão "+getarg(1) * .@amount+" Zeny.";
	mes "Você deseja continuar?";
	next;
	if (select("Sim:Não") == 1) {
		if (Zeny < getarg(1) * .@amount) {
			mes "[Vendedor de Livros de Alquímia]";
			mes "Você não possui";
			mes "zenys suficientes.";
			mes "Verifique quanto você possui";
			mes "em zeny primeiramente.";
			close;
		}
		if (checkweight(getarg(0),.@amount) == 0){
			mes "[Vendedor de Livros de Alquímia]";
			mes "Parece que você não pode carregar nada.";
			mes "Por favor verifique seu inventário.";
			close;
		}
		mes "[Vendedor de Livros de Alquímia]";
		mes "Obrigado pela ajuda.";
		set Zeny,Zeny - getarg(1) * .@amount;
		getitem getarg(0),.@amount;
		close;
	}
	mes "[Vendedor de Livros de Alquímia]";
	mes "Por favor, pense um pouco";
	mes "antes de tomar sua decisão.";
	close;
}
