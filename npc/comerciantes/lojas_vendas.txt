// Mercado Negro (Compra de Licenças - Classes que não sejam Mercador e Evoluções)
//============================================================ 
que_job01,68,84,1	script	Mercado Negro#Compras	4_M_SITDOWN,{

	if (checkweight(1201,1) == 0 || MaxWeight - Weight < 2400) {
		mes "[Sr. Jass]";
		mes "Você é muito ganancioso, mesmo em comparação com alguém como eu!";
		mes "Porque não vai esvaziar seu inventário primeiro?";
		close;
	}
	if (getskilllv("ALL_BUYING_STORE") == 1) {
		mes "[Sr. Jass]";
		mes "Ei, você já fez um contrato com Hugh.";
		mes "Eu não tenho nenhum negócio com você.";
		close;
	}
	mes "[Sr. Jass]";
	mes "Você deve precisar muito de alguma coisa para vir me procurar.";
	mes "O que você precisa?";
	next;
	switch(select("Lincença de Grande Comprador de Lojas:Quem é você?:Nada, nada!")) {
	case 1:
		mes "[Sr. Jass]";
		mes "Eu sabia!";
		mes "Claro, eu posso fazer isso por você.";
		mes "Mine apenas se parece com os autênticos comerciantes que começa a partir desse canalha Sr. Hugh!";
		next;
		mes "[Sr. Jass]";
		mes "E a minha licença é melhor 'porque você não precisa' aprender todas as habilidades.";
		mes "Quantas você quer?";
		next;
		mes "[Sr. Jass]";
		mes "Só para você saber, eu só posso fazê-las em pequenas quantidades, até 10 de uma vez.";
		mes "Vai custar 500 zeny para cada um.";
		next;
		while(1) {
			input .@input;
			mes "[Sr. Jass]";
			if (.@input == 0) {
				mes "Você não precisa dessas licenças?";
				close;
			} else if (.@input > 10) {
				mes "Eu só posso fazer até 10 de cada vez, você sabe.";
				next;
			} else {
				mes "Vai custar "+(.@input*500)+" zeny.";
				if (Zeny < (.@input*500)) {
					mes "mas você não tem dinheiro suficiente.";
					next;
					mes "[Sr. Jass]";
					mes "Você não sabe o básico de negócios? Tudo tem um preço.";
					mes "Se você quer algo, você tem que pagar por isso.";
				} else {
					mes "Ha... Ha ha ha!";
					mes "Sr. Hugh, Eu vou assumir a sua licença comercial. Você vai ver!";
					mes "*Risos Risos*";
					getitem 12548,.@input; // Licença Falso Atacado
					set Zeny,Zeny-(.@input*500);
				}
				close;
			}
		}
	case 2:
		mes "[Sr. Jass]";
		mes "Deixei minha cidade natal há muito tempo.";
		mes "Faz sentido perguntar quem eu sou, porque tudo o que eu tive deixado para trás até agora foi meu ódio";
		next;
		mes "[Sr. Jass]";
		mes "...";
		mes "Hugh é um comerciante corrupto sem nenhum senso de ética nos negócios.";
		mes "Meu único objetivo na vida é destruir Hugh.";
		next;
		mes "[Sr. Jass]";
		mes "Ah, eu bebi demais... (*Soluço*)";
		mes "Isso é só o álcool a falar.";
		mes "Por favor, esqueça tudo o que eu disse.";
		close;
		case 3:
		mes "[Sr. Jass]";
		mes "Certo, certo! Você não precisa gritar.";
		mes "Me deixe em paz se você não tem negócio comigo.";
		close;
	}
}

			mes "Você está certo, mas pense comigo:";
			mes "Você não teve dificuldade de comprar em massa?";
			mes "Você tem que encontrar e conversar com todo mundo que tem um item que você deseja.";

			mes "[Sr. Hugh]";
			mes "'Vamos Abrir Individualmente";
			mes "Uma Loja de Muitos Compradores!'";
			mes "Esse é o título da minha proposta.";
			mes "Você vai ver, quando você ler isso...";
			next;
			mes "[Sr. Hugh]";
			mes "...";
			mes "...(Sr. Hugh falou-me muito, nos mínimos detalhes.)";
			next;
			select("Certo, onde quer chegar?");
			mes "[Sr. Hugh]";
			mes "Ah, sim. Em resumo,";
			mes "você pode comprar alguns itens em massa através de Lojas.";

			mes "[Sr. Hugh]";
