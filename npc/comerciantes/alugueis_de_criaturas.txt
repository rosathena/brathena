// -------------- Criador de Pecopecos (para Cavaleiros e evoluções) --------------
prontera,55,350,5	script	Criador de Pecopecos#cav	8W_SOLDIER,{
	if (Upper >= 0 && Upper <= 2) set .@price,2500; // Pecopeco Normal - Avançado e Bebê

	mes "[Criador de Pecopecos]";
	if (BaseJob == Job_Knight && Class < Job_Rune_Knight) {
		mes "Bom-dia honorável Cavaleiro gostaria de comprar um Pecopeco?";
		mes "Você precisa ter a Habilidade Montaria e apenas "+.@price+" Zeny sobrando.";
		next;
		if (select("Alugar um Pecopeco:Cancelar"))
		if (@menu == 1) {
			if (Zeny < .@price || !getskilllv("KN_RIDING") || checkriding() || ismounting()) {
				mes "[Criador de Pecopecos]";
				mes ""+((Zeny < .@price) ? "Você não possui Zeny suficiente. Você está... pobre?" : (!getskilllv("KN_RIDING")) ? "Me desculpe, mas você não pode utilizar esse serviço. Por favor aprenda a habilidade Montaria primeiro." : "Você já está montado em uma montaria.")+"";
			} else {
			set Zeny, Zeny-.@price;
			setriding;
			}
			close;
			}
			mes "[Criador de Pecopecos]";
			mes "Entendo.";
			mes "Bem então, tenha um bom dia.";
			close;
		}
		mes "Me desculpe, mas esses Peco Pecos estão disponíveis";
		mes "apenas para Cavaleiros e Lordes.";
		close;
}
