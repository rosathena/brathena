gonryun,180,118,4	script	Homem de Ferro#gnp	4_M_03,{
	if (checkweight(1201,1) == 0) {
		mes "- Espere um minuto !! -";
		mes "- Atualmente você está muito pesado -";
		mes "- com vários itens. -";
		mes "- Por favor, tente novamente -";
		mes "- após perder peso. -";
		close;
	}
	if (BaseJob == Job_Novice) {
		mes "[Songmoodoo]";
		mes "Crianças não são permitidas aqui.";
		close;
	}
	if (in_battle == 3) set in_battle,1;
	if (in_battle < 1) {
		mes "[Songmoodoo]";
		mes "Ei ei~ Isso não é algo";
		mes "que se vê todos os dias.";
		mes "Ah~ você parece ser muito forte!";
		mes "Quer experimentar?";
		next;
		switch(select("Sim porque não?:?????:Não obrigado.")){
		case 1:
			mes "[Songmoodoo]";
			mes "Haha, eu sabia que você ia tentar.";
			mes "Se você conseguir levantar aquilo,";
			mes "eu lhe direi algo interessante.";
			mes "Pronto?";
			next;
			mes "^0000FFSongmoodoo aponta para a rocha.";
			mes "e pede para você levanta-la.";
			if (BaseJob < Job_Knight) callsub L_Lifting,690;
			else if (BaseClass == Job_Swordman || BaseClass == Job_Thief || BaseClass == Job_Merchant || BaseJob == Job_Monk) callsub L_Lifting,1100;
			else callsub L_Lifting,730;
			break;
		case 2:
			mes "[Songmoodoo]";
			mes "Essa não é uma boa idéia";
			mes "Mas se você acredita ser forte,";
			mes "essa é uma grande oportunidade para tentar";
			mes "algo novo. Quer tentar?";
			break;
		case 3:
			mes "[Songmoodoo]";
			mes "Também acho que não...";
			emotion e_pif;
			break;
		}
	} else if (in_battle == 1) {
		mes "[Songmoodoo]";
		mes "É bom ver você novamente!";
		mes "Vamos lá?";
		next;
		if (select("Claro:Talvez na próxima") == 1) {
			mes "[Songmoodoo]";
			mes "Muito bom! Haha.";
			mes "Eu gosto de sua confiança.";
			mes "Boa sorte para você~";
			close2;
			warp "gon_test",53,6;
			end;
		} else {
			mes "[Songmoodoo]";
			mes "Bem... certo.";
			mes "Te vejo numa próxima.";
		}
	} else {
		mes "[Songmoodoo]";
		mes "Você correu até lá?";
		mes "Você acha que não é forte o suficiente?";
		mes "Eu lhe darei outra chance.";
		mes "Até mais novamente!";
		emotion e_pif;
		set in_battle,1;
	}
	close;

L_Lifting:
	if (checkweight(7049,getarg(0))) {
		mes "Você levantou a pedra de forma leve.^000000";
		set in_battle,1;
		getitem 7049, getarg(0); // Pedra
		next;
		emotion e_no1;
		mes "[Songmoodoo]";
		mes "Uau! Excelente!";
		mes "Levarei você para um lugar legal";
		mes "na próxima vez que eu ver você.";
		mes "Até mais!";
	} else {
		mes "Parece muito pesado para você.^000000";
		next;
		mes "[Songmoodoo]";
		mes "Você ainda não está bem treinado.";
		mes "Volte quando tiver mais força.";
	}
	return;
}
