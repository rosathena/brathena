sec_in02,72,180,3	script	Administrador da Arena	4_NFWISP,{
	set .@i, callfunc("F_GM_NPC",1357,0);
	if (.@i == -1) {
		mes "[Administrador da Arena]";
		mes "O comando foi cancelado.";
		close;
	} else if (.@i == 0) {
		mes "Senha Incorreta";
		close;
	} else {
		mes "[Administrador da Arena]";
		mes "Selecione uma opção.";
		next;
		switch(select("Reiniciar a Arena","Reorganizar o Tempo de Ranking")) {
		case 1:
			switch(select("Nível 50:Nível 60:Nível 70:Nível 80:Modo Grupo")) {
			case 1:
				set .@arena,50;
				break;
			case 2:
				set .@arena,60;
				break;
			case 3:
				set .@arena,70;
				break;
			case 4:
				set .@arena,80;
				break;
			case 5:
				mes "[Administrador da Arena]";
				mes "== Atenção ==";
				mes "Você optou por reiniciar a arena.";
				mes "Deseja continuar?";
				next;
				switch(select("Sim:Não")) {
				case 1:
					donpcevent "Ponox::OnStart";
					mes "[Administrador da Arena]";
					mes "A Arena foi reiniciada com sucesso.";
					close;
				case 2:
					mes "[Administrador da Arena]";
					mes "O comando foi cancelado.";
					close;
				}
			}
			mes "[Administrador da Arena]";
			mes "== Atenção ==";
			mes "Você escolheu reiniciar a arena de nível "+.@arena+".";
			mes "Deseja continuar?";
			next;
			switch(select("Sim:Não")) {
			case 1:
				donpcevent "Lv "+.@arena+"s Waiting Room::OnStart";
				mes "[Administrador da Arena]";
				mes "A arena foi reativada com sucesso.";
				close;
			case 2:
				mes "[Arena Manager]";
				mes "O comando foi cancelado.";
				close;
			}
		case 2:
			mes "[Administrador da Arena]";
			mes "== Atenção ==";
			mes "^CE0000Você optou por reorganizar o tempo de classificação. Certifique-se que esta não é uma boa decisão!^000000";
			next;
			mes "[Administrador da Arena]";
			mes "Tem certeza que quer reorganizar o tempo de ranking?";
			next;
			switch(select("Não:Sim" )) {
			case 1:
				mes "[Administrador da Arena]";
				mes "Phew~ :)";
				close;
			case 2:
				switch(select("Lv 50:Lv 60:Lv 70:Lv 80:Party Mode")) {
				case 1:
					set .@min$,"$top50min";
					set .@sec$,"$top50sec";
					set .@mode$,"Lv 50";
					break;
				case 2:
					set .@min$,"$top60min";
					set .@sec$,"$top60sec";
					set .@mode$,"Lv 60";
					break;
				case 3:
					set .@min$,"$top70min";

					set .@mode$,"Lv 70";
					break;
				case 4:
					set .@min$,"$top80min";

					set .@mode$,"Lv 80";
					break;

					set .@min$,"$top_ptmin";

					break;
				}

function	script	Func_Are_Rew	{
	mes "[Givu]";
	mes "Gostaria de trocar os seus pontos de arena por "+getitemname(getarg(0))+"?";
	mes "Você pode trocar ^3131FF"+getarg(2)+" pontos de arena por "+getarg(1)+" "+getitemname(getarg(0))+"^000000.";
	mes "Se você quiser cancelar, por favor, digite 0. Se não, por favor insira quantos ^3131FFpontos^000000 você deseja gastar.";
	next;
	input .@reward;
	if (!.@reward) {
		mes "[Givu]";
		mes "Você cancelou seu pedido.";
		close;
	} else if (.@reward > 1500) {
		mes "[Givu]";
		mes "Você excedeu a capacidade máxima.";
		close;
	}
	if (arena_point < .@reward * getarg(2)) {
		mes "[Givu]";
		mes "Você não tem pontos de arena suficiente.";
		mes "Por favor, verifique a quantidade total de pontos de arena que você tem.";
		close;
	}
	set arena_point,arena_point - (.@reward * getarg(2));
	getitem getarg(0),.@reward * getarg(1);
	mes "[Givu]";
	mes "Obrigado, por favor, volte.";
	close;
}
