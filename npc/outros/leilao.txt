moc_ruins,78,173,6	script	Guia do Leilão#moc	4W_M_02,{
	if (strnpcinfo(0) == "Guia do Leilão#moc")
	callsub LeiTeleporte,1;
	else if (strnpcinfo(0) == "Guia do Leilão#prt")
	callsub LeiTeleporte,2;
	else if (strnpcinfo(0) == "Guia do Leilão#yuno")
	callsub LeiTeleporte,3;
	else if (strnpcinfo(0) == "Guia do Leilão#lhz")
	callsub LeiTeleporte,4;
	
	LeiTeleporte:
	mes "[Guia do Leilão]";
	mes "Olá, você gostaria";
	mes "de entrar na sala do";
	mes "Leilão?";
	next;
	if (select("Sim:Não") == 1) {
			mes "[Guia do Leilão]";
		if (getarg(0) == 1 || getarg(0) == 4) {
			mes "Certo! Espero que você";
			mes "se divirta muito e";
			mes "aproveite o leilão~";
		} else {
			mes "Tenha um bom proveito no leilão.";
		}
		close2;
		switch(getarg(0)) {
		case 1:
			warp "auction_01",179,53; end;
		case 2:
			warp "auction_01",21,43; end;
		case 3:
			warp "auction_02",151,23; end;
		case 4:
			warp "auction_02",43,24; end;
		}
	}
	mes "[Guia do Leilão]";
	mes "Tudo bem, mas se";
	mes "depois você";
	mes "mudar de idéia";
	mes "por favor, venha";
	mes "e aproveite o leilão~";
	close;
	
	
	
}
