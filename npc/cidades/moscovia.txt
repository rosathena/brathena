alberta,246,51,4	script	Oficial de Moscovia#1	4_F_RUSWOMAN2,{
	mes "[Oficial de Moscovia]";
	mes "Moscovia! Você conhece Moscovia?";
	mes "O paraíso se espalhando nos mares sem fim...";
	mes "Bem-vindo a Moscovia.";
	mes "É uma terra mística.";
	next;
	switch(select("Sobre Moscovia...:Ir para Moscovia.:Cancelar")) {
	case 1:
		mes "[Oficial de Moscovia]";
		mes "Moscovia é um lindo reino, uma ilha localizada no norte de Rune-Midgard.";
		next;
		mes "[Oficial de Moscovia]";
		mes "Sei que você ficará fascinado pelo lindo cenário de Moscovia, e pelo belo palácio.";
		next;
		mes "[Oficial de Moscovia]";
		mes "Agora que nosso longo inverno passou, estou feliz por poder mostrar a você nossa linda cidade.";
		next;
		mes "[Oficial de Moscovia]";
		mes "Se você quiser visitar Moscovia, vá agora!";
		mes "Eu levarei você em uma boa viagem por Moscovia!";
		close;

	case 2:
		mes "[Oficial de Moscovia]";
		mes "Ok, vamos começar agora.";
		mes "Você deve me pagar 10 000 zenys para ir a Moscovia.";
		mes "Mas não precisa pagar para voltar.";
		next;
		mes "[Oficial de Moscovia]";
		mes "Podemos ir?";
		next;
		if (select("Vamos!:Cancelar") == 2) {
			mes "[Oficial de Moscovia]";
			mes "Se você está ocupado agora, me avise quando quiser ir.";
			mes "Estou sempre pronto para guiar alguém por Moscovia.";
			close;
		}
			mes "[Oficial de Moscovia]";
      if (Zeny < 10000) {
			mes "Sinto muito, mas você não tem dinheiro suficiente.";
			mes "Precisa de 10 000 zenys para ir a Moscovia.";
			mes "Obrigado.";
			close;
      }
			mes "Certo, estamos indo agora.";
			close2;
			set Zeny,Zeny - 10000;
			warp "moscovia",163,55;
			end;
	case 3:
		mes "[Oficial de Moscovia]";
		mes "Se você está ocupado agora, me avise quando quiser ir.";
		mes "Estou sempre pronto para guiar alguém por Moscovia.";
		close;
	}
}

	mes "Tem boas lembranças de Moscovia?";

	next;
	if (select("Retornar para Alberta:Cancelar") == 2) {
		mes "[Oficial de Moscovia]";
		mes "Se quiser passear mais, leve o tempo que quiser.";
		close;
	}
	mes "[Oficial de Moscovia]";
	mes "Por favor venha vistar-nos.";
	mes "Certo, vamos indo.";
	close2;
	if (checkversion(0))
		warp "alberta",244,52;
	else
		warp "alberta",243,67;
	end;
}
