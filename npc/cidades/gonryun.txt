alberta,245,69,3	script	Barqueiro de Kunlun#gon	4_M_TWMIDMAN,{
	mes "[Wabakho]";
	mes "Bom dia~";
	mes "Deixem-me convidá-los para visitar a minha terra natal, Kunlun.";
	mes "É uma honra guiar convidados tão distintos, naturais do reino de Rune-Midgard";
	next;
	switch(select("Sobre Kunlun:Visitar Kunlun:Cancelar")) {
	case 1:
		mes "[Wabakho]";
		mes "Kunlun é um lugar maravilhoso, com uma história rica e tradições próprias. Acredito também que Kunlun é o melhor lugar para se apreciar uma bela vista.";
		next;
		mes "[Wabakho]";
		mes "Quando você chegar no porto de Kunlun, poderá ver réplicas em miniatura de prédios de Alberta e Prontera.";
		next;
		mes "[Wabakho]";
		mes "Depois de aproveitar um bom passeio, passe pela bonita coluna de luz que te levará acima das nuvens, até a vila de Kunlun.";
		next;
		mes "[Wabakho]";
		mes "Já ouvi dizer que no continente de Rune-Midgards existe outra cidade que é mantida suspensa no ar por um poder antigo e misterioso...";
		next;
		mes "[Wabakho]";
		mes "Bem, Kunlun também flutua no ar, mas sem nenhuma tecnologia ou fonte de energia.";
		mes "Consideramos que nossa terra é especialmente abençoada...";
		next;
		mes "[Wabakho]";
		mes "Quando estiver em Kunlun, não deixe de experimentar nossas especialidades como o pêssego do céu.";
		next;
		mes "[Wabakho]";
		mes "E é melhor você se preparar se estiver planejando uma visita à dungeon de Kunlun.";
		mes "Não é um lugar seguro!";
		next;
		mes "[Wabakho]";
		mes "Se você está interessado em visitar Kunlun, venha ogo me dizer. É um prazer serví-lo, honroso convidado.";
		close;
	case 2:
		mes "[Wabakho]";
		mes "Excelente escolha! Estou feliz por tê-lo como convidado~";
		mes "Entretanto, uma pequena taxa é requerida para embarcar.";
		next;
		mes "[Wabakho]";
		mes "Pedimos que paque 10.000 zenys antes de viajar. Essa taxa já inclui seu retorno a Alberta.";
		mes "Estou pronto para levá-lo a Kunlun quando quiser.";
		next;
		mes "[Wabakho]";
		mes "Você gostaria de embarcar?";
		next;
		if (select("Para Kunlun~!:Não.") == 1) {
			if (Zeny > 9999) {
				mes "[Wabakho]";
				mes "Obrigado, levarei você imediatamente.";
				close2;
				set Zeny, Zeny-10000;
				warp "gon_fild01",258,82;
				end;
			}
			mes "[Wabakho]";
			mes "Sinto muito, você precisa de 10 000 zenys para viajar à Kunlun.";
			mes "Certifique-se de ter dinheiro suficiente em suas mãos. Obrigado e retorne quando quiser!";
			close;
		}
		mes "[Wabakho]";
		mes "Entendo. Mas, se mudar de ideia, por favor me avise.";
		mes "Seria um enorme prazer servi-lo, honroso viajante.";
		close;
	case 3:
		mes "[Wabakho]";
		mes "Entendo. Entretanto, se mudar de ideia, me avise.";
		mes "Será um prazer serví-lo.";
		close;
	}
}

	mes "Provavelmente ainda existem muitos lugares para você visitar.";
	close;
}
