alberta,246,29,3	script	Aibakthing#ayo	4_M_THAIONGBAK,{
	mes "[Aibakthing]";
	mes "Hoo! Hah! Hoo! Hah!";
	mes "Deixe-me levar-te para um lugar espiritual distante, uma terra de mistério exótico, minha cidade natal Ayothaya!";
	next;
	switch(select("Sobre Ayothaya.:Ir para Ayothaya.:Cancelar.")) {
	case 1:
		mes "[Aibakthing]";
		mes "Nós, moradores de Ayothaya, somos pessoas piedosas que damos valor à paz e sinceridade. Não nos esforçamos pra levar uma vida simples, porém nobre, em harmonia com a natureza.";
		next;
		mes "[Aibakthing]";
		mes "A nossa cozinha tradicional é mundialmente famosa. Não estranhe quando servirem pratos como Tom Yum Goong, feito de Camarão, Limão e Chilli, que tem um aroma e sabor além da imaginação.";
		next;
		mes "[Aibakthing]";
		mes "Para aqueles que tem fome e sede de aventura, não existe história mais antiga sobre Sa-mhing Tiger, uma criatura que assombra as ruínas do nosso templo.";
		next;
		mes "[Aibakthing]";
		mes "Hoo! Hahh! Hoo! Hahh!";
		mes "Nenhum homem vive pode resistir ao convite desta bela terra.";
		mes "Habitantes de toda Rune-Midgard são sempre bem-vindos!";
		close;
	case 2:
		mes "[Aibakthing]";
		mes "Ah, você tem que pegar 10.000 Zeny se quiser visitar Ayothaya.";
		mes "Quando estiver pronto, poderemos ir!";
		next;
		if (select("Estou pronto, vamos lá!:Não.") == 1) {
				mes "[Aibakthing]";
			if (Zeny > 9999) {
				mes "Hoo! Hah! Hoo! Hah!";
				mes "Vamos partir! Voltar para a minha bela Ayothaya!";
				close2;
				set Zeny, Zeny-10000;
				warp "ayothaya",149,71;
				end;
			}
			mes "Lamento, mas você não possui 10.000 zeny para viajar para Ayothaya. Tal preço não é nada comparado com as experiências que esperam por você!";
			close;
		}
		break;
		
		case 3:
		mes "[Aibakthing]";
		mes "Alguma vez você já sonhou com um belo lugar cheio de serenidade espiritual? Você deve ter sonhado com Ayothaya, meu amigo.";
		close;
	}
}
