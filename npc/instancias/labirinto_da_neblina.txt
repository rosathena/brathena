1@mist,0,0,0	script	mist#spawn_mvp	-1,{
OnInstanceInit:
	donpcevent instance_npcname("mist#spawn_mvp")+"::OnEnable";
	end;
OnDisable:
	killmonster instance_mapname("1@mist"),instance_npcname("mist#spawn_mvp")+"::OnMyMobDead";
	end;
	
OnEnable:
	enablenpc instance_npcname("mist#spawn_mvp");
	set .@mob_ran,rand(1,30);
	if (.@mob_ran < 6)
		areamonster instance_mapname("1@mist"),133,155,144,151,"Dragão Púrpura",2131,1,instance_npcname("mist#spawn_mvp")+"::OnMyMobDead";
	else if ((.@mob_ran > 5) && (.@mob_ran < 11))
		areamonster instance_mapname("1@mist"),256,180,268,173,"Dragão Púrpura",2131,1,instance_npcname("mist#spawn_mvp")+"::OnMyMobDead";
	else if ((.@mob_ran > 10) && (.@mob_ran < 16))
		areamonster instance_mapname("1@mist"),54,290,68,278,"Dragão Púrpura",2131,1,instance_npcname("mist#spawn_mvp")+"::OnMyMobDead";
	else if ((.@mob_ran > 15) && (.@mob_ran < 21))
		areamonster instance_mapname("1@mist"),244,227,258,216,"Dragão Púrpura",2131,1,instance_npcname("mist#spawn_mvp")+"::OnMyMobDead";
	if ((.@mob_ran > 20) && (.@mob_ran < 26))
		areamonster instance_mapname("1@mist"),183,267,196,260,"Dragão Púrpura",2131,1,instance_npcname("mist#spawn_mvp")+"::OnMyMobDead";
	if (.@mob_ran > 26)
		areamonster instance_mapname("1@mist"),176,311,189,304,"Dragão Púrpura",2131,1,instance_npcname("mist#spawn_mvp")+"::OnMyMobDead";
OnMyMobDead:
	set .@callnum,1;
	set .@v_mob_num,mobcount("1@mist",instance_npcname("mist#spawn_mvp")+"::OnMyMobDead");
	set .@mob_dead_num,.@callnum - .@v_mob_num;
	if (.@mob_dead_num > 0) {
		disablenpc instance_npcname("Jovem Misterioso#npc_mist3");
		enablenpc instance_npcname("Loki#npc_mist2");
		enablenpc instance_npcname("Garota Desmaiada#npc_mist1");
		mapannounce instance_mapname("1@mist"), "Dragão Púrpura: Estou sendo punido... por ter abandonado... meu caminho de Guardião?",bc_map,"0xFFFF00";
		disablenpc instance_npcname("mist#spawn_mvp");
	}
	end;
