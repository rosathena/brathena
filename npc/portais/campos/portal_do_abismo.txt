	if ((countitem(1035)) && (countitem(1036)) && (countitem(1037))) {
		next;
		switch(select("Inserir partes do Corpo do Dragão","Comer Partes do Corpo do Dragão" ) ) {
			case 1:
				mes "^3355FFVocê coloca cuidadosamente o";

				next;
				mes "^3355FFVocê coloca lentamente";
				mes "a Escama de Dragão em uma outra";
				mes "rachadura da coluna,";
				mes "causando outro pequeno";
				mes "tremor vindo da coluna.^000000";
				specialeffect EF_WINDHIT;
				next;
				mes "^3355FFVocê cautelosamente insere";
				mes "a Cauda de Dragão dentro da";
				mes "rachadura final. Luzes brilhantes";
				mes "surjem de dentro do";
				mes "interior das colunas...^000000";
				specialeffect EF_WINDHIT;
				next;
				mes "^3355FFO solo abaixo de";
				mes "seus pés começa a";
				mes "tremer violentamente.^000000";
				specialeffect EF_BOWLINGBASH;
				donpcevent "Portal do Abismo::OnWarp";

				delitem 1035,1;
				delitem 1036,1;
				delitem 1037,1;
				close2;

			case 2:
				mes "^3355FFVocê cautelosamente coloca";
