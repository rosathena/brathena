bat_a01,15,15,3	script	start#bat_a01	CLEAR_NPC,{
OnInit:
	mapwarp "bat_a01","bat_room",154,150;
	end;

OnEnable:
	setarray .@doev$,
	"OBJ#bat_a01_a::OnKill","OBJ#bat_a01_a::OnEnable","OBJ#bat_a01_b::OnKill","OBJ#bat_a01_b::OnEnable","barricade#bat_a01_a::OnKill",
	"barricade#bat_a01_a::OnEnable","barricade#bat_a01_b::OnKill","barricade#bat_a01_b::OnEnable","OBJ#bat_a01_n::OnKill","OBJ#bat_a01_n::OnEnable",
	"NOBJ_mob#bat_a01_a::OnKill","NOBJ_mob#bat_a01_b::OnKill","Terapeuta de Batalha#a01_a::OnEnable","Terapeuta de Batalha#a01_b::OnEnable","countdown#bat_a01::OnEnable";
	
	setarray .@disb$,"Ferreiro Guillaume#a01","Ferreiro Croix#bat_a01","Vintenar Guillaume#a01_a","Vintenar Croix#a01_b";
	
	for(set .@i, 0; .@i < getarraysize(.@doev$); set .@i, .@i+1)
		donpcevent .@doev$[.@i];
		
	for(set .@i, 0; .@i < getarraysize(.@disb$); set .@i, .@i+1)
		disablenpc .@disb$[.@i];
	
	end;

OnGuillaumeQuit:
OnCroixQuit:
	bg_leave;
	end;

OnReadyCheck:
	if ( $@TierraBG1 )
		end;
	set .@Guillaume, getwaitingroomstate(0,"Tenente Kalos");
	set .@Croix, getwaitingroomstate(0,"Tenente Eyor");
	if ( !.@Guillaume && !.@Croix ) {
		donpcevent "#bat_a01_timer::OnStop";
		end;
	}
	else if ( .@Guillaume < 10 || .@Croix < 10 )
		end;
	set $@TierraBG1,1;
	donpcevent "Tenente Kalos::OnEnterBG";
	donpcevent "Tenente Eyor::OnEnterBG";
	donpcevent "start#bat_a01::OnEnable";
	bg_warp $@TierraBG1_id1,"bat_a01",352,342;
	bg_warp $@TierraBG1_id2,"bat_a01",353,52;
	initnpctimer;
	end;

OnTimer10000:
	stopnpctimer;
	donpcevent "#bat_a01_timer::OnEnable";
	end;
}
