			mes "[Vendedor de Ovos]";
			mes "Ótimo! Quantos você";

			mes "Lembre-se que você não vai";
			mes "sempre conseguir comprar";
			mes "ovos de galinha tão facilmente!";

					mes "você se importaria de ficar";
					mes "fora do caminho para que eu possa";
					mes "a ajudar os meus clientes?";
					close;

					mes "[Vendedor de Ovos]";
					mes "Ei, você mente, dando";
					mes "um número desse? Eu também";

					mes "quantos que deseja de verdade agora.";
					next;
				}
				else {
					set .@egg_zeny,.@input*500;
					if (Zeny < .@egg_zeny) {
						mes "[Vendedor de Ovos]";
						mes "Você não tem Zeny suficiente.";
						next;
						mes "[Vendedor de ovos]";
						mes "Nenhum desconto, vai ter Zeny o suficiente.";
						close;
					}
					else {
						if (checkweight(7605,.@input) == 0) {
							mes "[Vendedor de Ovos]";
							mes "Ei, eu sei que você realmente";
							mes "quer comprar alguns ovos";
							mes "de galinha, mas você vai ser capaz";
							mes "para levar tantos ovos agora. Por que";
							mes "você não liberar algum";
							mes "espaço no inventário primeiro?.";
							close;
						}
						else {
							mes "[Vendedor de Ovos]";
							mes "Aqui está seus ovos de galinha!";
							mes "Eu acho que há algum festival";
							mes "acontecendo, onde você precisa";
							mes "deles, mas eu não sei de nada";
							mes "sobre isso. Bem, tem um bom";
							mes "dia, e espero vê-lo de novo!";
							set Zeny, Zeny-.@egg_zeny;
							getitem 7605,.@input; //Chicken_Egg
							close;

					}
				}
			}
		case 4:
			mes "[Vendedor de Ovos]";
			mes "Não esta interessado em comprar";
			mes "Ovos de galinha, hein? Bem,";

		next;

		mes "^4D4DFF1 Erva Verde^000000,";

		mes "^4D4DFF1 Água Benta^000000, e";
		mes "^4D4DFF1 Ovos de galinha^000000.";
		close;
	case 2:
		mes "[Nerlen]";
		mes "Tudo bem, quantos";
		mes "Ovos Sagrados você";
		mes "quer fazer?";
		mes "Eu só posso fazer até";
		mes "10 Ovos Sagrados de cada vez.";
		mes "Digite ''0'' para cancelar.";
		next;
		while(1) {
			input .@input;
			if (.@input == 0) {
				mes "[Nerlen]";
				mes "Oh, você mudou a sua";
				mes "ídeia? volte quando quiser,";
				mes "sempre que você quiser que eu";
				mes "faça alguns Ovos Sagrados ~";
				close;
			}
			else if (.@input < 0 || .@input > 10) {
				mes "[Nerlen]";
				mes "Me desculpe, mas eu só posso";
				mes "fazer até 10 Ovos Sagrados";
				mes "ao mesmo tempo. Se você quisesse";
				mes "Para cancelar digite ''0''";
				next;
			}
			else {
				if ((countitem(509) < .@input) || (countitem(511) < .@input) || (countitem(508) < .@input) || (countitem(523) < .@input) || (countitem(7605) < .@input)) {
					mes "[Nerlen]";
					mes "Me desculpe, mas você não";
					mes "tem elementos suficientes para fazer";
					mes "Ovos Sagrados. Você precisa do";
					mes "seguinte para que eu consiga criar";
					mes "Cada Ovo Sagrado para você.";
					next;
					mes "[Nerlen]";
					mes "^4D4DFF1 Erva Branca^000000,";
					mes "^4D4DFF1 Erva Verde^000000,";
					mes "^4D4DFF1 Erva Amarela^000000,";
					mes "^4D4DFF1 Água Benta^000000, e";
					mes "^4D4DFF1 Ovos de galinha^000000.";
					close;
				}
				else {
					break;
				}
			}
		}
		if (checkweight(12019,.@input) == 0) {
			mes "[Nerlen]";
			mes "Me desculpe, mas você";
			mes "não será capaz de transportar";
			mes "todos Ovos Sagrados agora ...";
			mes "É melhor liberar mais";
			mes "espaço em seu inventário primeiro.";
			close;
		}
		else {
			delitem 509,.@input; //White_Herb
			delitem 511,.@input; //Green_Herb
			delitem 508,.@input; //Yellow_Herb
			delitem 523,.@input; //Holy_Water
			delitem 7605,.@input; //Egg
			getitem 12019,.@input; //Holy_Egg
			mes "[Nerlen]";
			mes "Obrigado!";
			mes "Espero que gostem";
			mes "estes Ovos Sagrados~";
			mes "Espero que deus esteja com você.";
			close;
		}
	case 3:
		mes "[Nerlen]";
		mes "Tudo bem, eu entendo.";
		mes "Que Deus os abençoe.";
		close;
	}
}
