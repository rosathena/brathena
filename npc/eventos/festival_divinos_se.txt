		mes "Senha incorreta.";

	mes "[Gerente do Festival]";

	next;

		donpcevent "Rmimi Ravies#gq_fes01::OnOver";
		donpcevent "Rmimi Ravies#gq_fes01::OnOver";
		close;
	case 2:
		mes "[Gerente do Festival]";
		mes "Redefinindo o festival e ativando os NPCs.";
		donpcevent "Rmimi Ravies#gq_fes01::OnOver";
		donpcevent "Rmimi Ravies#gq_fes01::OnEnable";
		close;
	}
}
